CREATE SCHEMA IF NOT EXISTS `testpl` DEFAULT CHARACTER SET utf8 COLLATE utf8_polish_ci ;
USE `testpl` ;

-- -----------------------------------------------------
-- Table `testpl`.`Songs`
-- -----------------------------------------------------
CREATE TABLE IF NOT EXISTS `testpl`.`Songs` (
  `Id` INT NOT NULL AUTO_INCREMENT,
  `Name` VARCHAR(200) NOT NULL,
  `FileName` VARCHAR(200) NOT NULL,
  PRIMARY KEY (`Id`),
  UNIQUE INDEX `FileName_UNIQUE` (`FileName` ASC),
  UNIQUE INDEX `Name_UNIQUE` (`Name` ASC))
ENGINE = InnoDB;


-- -----------------------------------------------------
-- Table `testpl`.`MassPart`
-- -----------------------------------------------------
CREATE TABLE IF NOT EXISTS `testpl`.`MassPart` (
  `Id` INT NOT NULL AUTO_INCREMENT,
  `Name` VARCHAR(20) NOT NULL,
  `FullName` VARCHAR(45) NOT NULL,
  PRIMARY KEY (`Id`))
ENGINE = InnoDB;


-- -----------------------------------------------------
-- Table `testpl`.`SongToMassPart`
-- -----------------------------------------------------
CREATE TABLE IF NOT EXISTS `testpl`.`SongToMassPart` (
  `Id` INT NOT NULL AUTO_INCREMENT,
  `Songs_Id` INT NOT NULL,
  `MassPart_Id` INT NOT NULL,
  PRIMARY KEY (`Id`),
  INDEX `fk_table1_Songs_idx` (`Songs_Id` ASC),
  INDEX `fk_table1_SongFor1_idx` (`MassPart_Id` ASC),
  CONSTRAINT `fk_table1_Songs`
    FOREIGN KEY (`Songs_Id`)
    REFERENCES `testpl`.`Songs` (`Id`)
    ON DELETE NO ACTION
    ON UPDATE NO ACTION,
  CONSTRAINT `fk_table1_SongFor1`
    FOREIGN KEY (`MassPart_Id`)
    REFERENCES `testpl`.`MassPart` (`Id`)
    ON DELETE NO ACTION
    ON UPDATE NO ACTION)
ENGINE = InnoDB;


-- -----------------------------------------------------
-- Table `testpl`.`Composers`
-- -----------------------------------------------------
CREATE TABLE IF NOT EXISTS `testpl`.`Composers` (
  `Id` INT NOT NULL AUTO_INCREMENT,
  `Name` VARCHAR(45) NULL,
  `SurName` VARCHAR(45) NULL,
  PRIMARY KEY (`Id`))
ENGINE = InnoDB;


-- -----------------------------------------------------
-- Table `testpl`.`SongToComposer`
-- -----------------------------------------------------
CREATE TABLE IF NOT EXISTS `testpl`.`SongToComposer` (
  `Id` INT NOT NULL AUTO_INCREMENT,
  `Songs_Id` INT NOT NULL,
  `Composers_Id` INT NOT NULL,
  `ManyVoice` BIT NOT NULL DEFAULT 0,
  PRIMARY KEY (`Id`),
  INDEX `fk_SongToComposer_Songs1_idx` (`Songs_Id` ASC),
  INDEX `fk_SongToComposer_Composers1_idx` (`Composers_Id` ASC),
  CONSTRAINT `fk_SongToComposer_Songs1`
    FOREIGN KEY (`Songs_Id`)
    REFERENCES `testpl`.`Songs` (`Id`)
    ON DELETE NO ACTION
    ON UPDATE NO ACTION,
  CONSTRAINT `fk_SongToComposer_Composers1`
    FOREIGN KEY (`Composers_Id`)
    REFERENCES `testpl`.`Composers` (`Id`)
    ON DELETE NO ACTION
    ON UPDATE NO ACTION)
ENGINE = InnoDB;


-- -----------------------------------------------------
-- Table `testpl`.`Category`
-- -----------------------------------------------------
CREATE TABLE IF NOT EXISTS `testpl`.`Category` (
  `Id` INT NOT NULL AUTO_INCREMENT,
  `Name` VARCHAR(20) NULL,
  `FullName` VARCHAR(45) NULL,
  PRIMARY KEY (`Id`))
ENGINE = InnoDB;


-- -----------------------------------------------------
-- Table `testpl`.`SongToCategory`
-- -----------------------------------------------------
CREATE TABLE IF NOT EXISTS `testpl`.`SongToCategory` (
  `Id` INT NOT NULL AUTO_INCREMENT,
  `Category_Id` INT NOT NULL,
  `Songs_Id` INT NOT NULL,
  PRIMARY KEY (`Id`),
  INDEX `fk_SongToCategory_Category1_idx` (`Category_Id` ASC),
  INDEX `fk_SongToCategory_Songs1_idx` (`Songs_Id` ASC),
  CONSTRAINT `fk_SongToCategory_Category1`
    FOREIGN KEY (`Category_Id`)
    REFERENCES `testpl`.`Category` (`Id`)
    ON DELETE NO ACTION
    ON UPDATE NO ACTION,
  CONSTRAINT `fk_SongToCategory_Songs1`
    FOREIGN KEY (`Songs_Id`)
    REFERENCES `testpl`.`Songs` (`Id`)
    ON DELETE NO ACTION
    ON UPDATE NO ACTION)
ENGINE = InnoDB;
